// Initialize suggested questions function initializeSuggestedQuestions() { // Add initial questions initialQuestions.forEach(question => { const button = document.createElement('button'); button.className = 'suggested-question'; // Create categorized questions container (hidden initially) const categoriesContainer = document.createElement('div'); categoriesContainer.className = 'question-categories'; categoriesContainer.id = 'questionCategories'; button.className = 'suggested-question'; button.textContent = question; button.addEventListener('click', () => { chatInput.value = question; handleSendChat(); }); categoryDiv.appendChild(button); }); categoriesContainer.appendChild(categoryDiv); }); suggestedQuestionsContainer.appendChild(categoriesContainer); } // Toggle categorized questions visibility function toggleCategorizedQuestions() { const categoriesContainer = document.getElementById('questionCategories'); revealView.canEdit = false; revealView.canSaveAs = false; revealView.showRefresh = false; revealView.canCopyVisualization = false; } initializeDashboard(); initializeSuggestedQuestions(); initializePersistentSuggestions(); // Initialize persistent suggestions panel function initializePersistentSuggestions() { // Combine a selection of questions for quick access const quickSuggestions = [ ...initialQuestions.slice(0, 3), ...categorizedQuestions['Time-Series & Trends'].slice(0, 2), ...categorizedQuestions['Company Analysis'].slice(0, 2), ...categorizedQuestions['Geographic Analysis'].slice(0, 1) ]; quickSuggestions.forEach(question => { const chip = document.createElement('button'); chip.className = 'suggestion-chip'; chip.textContent = question; chip.title = question; // Show full text on hover chip.addEventListener('click', () => { chatInput.value = question; chatInput.focus(); }); suggestionsContent.appendChild(chip); }); // Toggle suggestions panel suggestionsHeader.addEventListener('click', () => { suggestionsContent.classList.toggle('collapsed'); suggestionsToggle.classList.toggle('collapsed'); }); } // Show persistent suggestions panel when conversation starts function showPersistentSuggestions() { suggestionsPanel.classList.add('visible'); } // Add chat message to UI function addChatMessage(role, content) { // Remove empty state if present const emptyState = chatMessages.querySelector('.empty-state'); if (emptyState) { emptyState.remove(); showPersistentSuggestions(); } const messageDiv = document.createElement('div'); messageDiv.className = 'chat-message'; const avatarDiv = document.createElement('div'); avatarDiv.className = `message-avatar ${role}-avatar`; const avatarIcons = { user: '👤', assistant: '🤖', system: '⚙️', log: '📝', error: '⚠️' }; avatarDiv.textContent = avatarIcons[role] || '💬'; const contentDiv = document.createElement('div'); contentDiv.className = 'message-content'; const roleDiv = document.createElement('div'); roleDiv.className = 'message-role'; roleDiv.textContent = role === 'assistant' ? 'AI Assistant' : role; const textDiv = document.createElement('div'); textDiv.className = `message-text ${role}`; textDiv.innerHTML = content; contentDiv.appendChild(roleDiv); contentDiv.appendChild(textDiv); messageDiv.appendChild(avatarDiv); messageDiv.appendChild(contentDiv); chatMessages.appendChild(messageDiv); // Auto-scroll to bottom chatMessages.scrollTop = chatMessages.scrollHeight; return textDiv; } // Create streaming message placeholder function createStreamingMessage() { const textDiv = addChatMessage('assistant', ''); textDiv.classList.add('streaming'); return textDiv; } // Load dashboard from JSON response async function loadDashboardFromResponse(dashboardJson) { try { const json = JSON.parse(dashboardJson); revealView.dashboard = await $.ig.RevealUtility.createDashboardFromJsonObject(json); } catch (e) { console.error('Failed to load dashboard:', e); addChatMessage('error', 'Failed to load dashboard: ' + e.message); } } // Handle send chat message async function handleSendChat() { if (!chatInput || !chatInput.value.trim()) return; const question = chatInput.value.trim(); chatInput.value = ''; chatInput.style.height = 'auto'; // Disable send button during processing sendButton.disabled = true; addChatMessage('user', question); // Reset streamed explanation for new message streamedExplanation = ''; let streamingMessageDiv = null; try { const stream = await client.ai.chat.sendMessage({ message: question, datasourceId: 'NorthwindDB', // Example datasource, adjust as needed dashboard: revealView.dashboard, stream: true, }); stream.on('progress', (message) => { // Show progress messages as log entries addChatMessage('log', message); }); stream.on('text', (chunk) => { // Handle streaming text chunks (like ChatGPT) streamedExplanation += chunk; // Create or update the streaming message div if (!streamingMessageDiv) { streamingMessageDiv = createStreamingMessage(); } // Update the content with parsed markdown streamingMessageDiv.innerHTML = marked.parse(streamedExplanation); chatMessages.scrollTop = chatMessages.scrollHeight; }); stream.on('error', (error) => { console.error('Chat error:', error); addChatMessage('error', error); }); const result = await stream.finalResponse(); // Mark the streaming div as complete if (streamingMessageDiv) { streamingMessageDiv.classList.remove('streaming'); streamingMessageDiv.classList.add('completed'); } // Handle case where no streaming occurred (fallback to final response) if (!streamingMessageDiv && result.explanation) { const htmlContent = marked.parse(result.explanation); addChatMessage('assistant', htmlContent); } if (result.error) { addChatMessage('error', result.error); } // Load dashboard if present if (result.dashboard) { await loadDashboardFromResponse(result.dashboard); } } catch (error) { addChatMessage('error', `Error: ${error.message}`); } finally { // Re-enable send button sendButton.disabled = false; chatInput.focus(); } } // Event listeners sendButton.addEventListener('click', handleSendChat); chatInput.addEventListener('keydown', (e) => { if (e.key === 'Enter' && !e.shiftKey) { e.preventDefault(); handleSendChat(); } }); // Auto-resize textarea chatInput.addEventListener('input', function() { this.style.height = 'auto'; this.style.height = Math.min(this.scrollHeight, 200) + 'px'; }); // Focus input on load chatInput.focus();
